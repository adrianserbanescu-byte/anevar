```python
def valideaza_profil(profil) -> list["Issue"]:
    """Avertismente de consistență a profilului de evaluare."""
    issues: list[Issue] = []
    if not profil.abordari_aplicabile:
        issues.append(Issue(nivel="blocheaza",
                            mesaj="Profilul nu are nicio abordare aplicabilă."))
    for cheie in profil.ponderi:
        if cheie not in profil.abordari_aplicabile:
            issues.append(Issue(nivel="avertisment",
                                mesaj=f"Ponderea „{cheie}" nu corespunde unei abordări aplicabile."))
    return issues
```
